Variables in JavaScript

Variables are containers for storing data values. In JavaScript, there are 3 ways to declare variables: var, let, and const.

const

The const declaration creates a read-only reference to a value. It doesn't mean the value it holds is immutable, just that the variable identifier cannot be reassigned.

const country = "USA";

console.log(PI); // Output: 3.14159
console.log(country); // Output: USA

‚ö†Ô∏è Important

Once you declare a variable with const, you cannot reassign it. This will cause an error: const x = 5; x = 10; // Error!

Variable Naming Rules

  • Variable names must begin with a letter, underscore (_), or dollar sign ($)
  • Variable names are case-sensitive (y and Y are different variables)
  • Variable names cannot contain spaces or special characters except _ and $
  • Variable names cannot be JavaScript reserved words

Loops in JavaScript

Loops are used to execute the same block of code multiple times.

for Loop

The for loop repeats a block of code as long as a specified condition is true.

for (let i = 0; i < 5; i++) {
    console.log("Iteration " + i);
}

while Loop

The while loop loops through a block of code as long as a specified condition is true.

let i = 0;
while (i < 5) {
    console.log("While iteration " + i);
    i++;
}

do...while Loop

The do...while loop is a variant of the while loop. This loop will execute the code block once, before checking if the condition is true, then it will repeat the loop as long as the condition is true.

let j = 0;
do {
    console.log("Do-while iteration " + j);
    j++;
} while (j < 5);

break and continue

Use break to exit a loop, and continue to skip an iteration.

// Break example
for (let k = 0; k < 10; k++) {
    if (k === 5) {
        break; // Exit the loop when k is 5
    }
    console.log(k);
}

// Continue example
for (let m = 0; m < 10; m++) {
    if (m % 2 === 0) {
        continue; // Skip even numbers
    }
    console.log(m); // Only prints odd numbers
}

Promises

A Promise is an object representing the eventual completion or failure of an asynchronous operation.

// Creating a Promise
const myPromise = new Promise((resolve, reject) => {
    // Async operation
    if (/* operation successful */) {
        resolve("Success!");
    } else {
        reject("Error!");
    }
});

// Using a Promise
myPromise
    .then(result => console.log(result))
    .catch(error => console.log(error));
promises.js 
// Promise that resolves after 2 seconds
function delayedResolve(message, delay = 2000) {
    return new Promise((resolve) => {
        setTimeout(() => {
            resolve(message);
        }, delay);
    });
}

// Promise that rejects after 1 second
function delayedReject(message, delay = 1000) {
    return new Promise((_, reject) => {
        setTimeout(() => {
            reject(new Error(message));
        }, delay);
    });
}

console.log("Starting promise chain...");

delayedResolve("First success!")
    .then(result => {
        console.log(result);
        return delayedResolve("Second success!");
    })
    .then(result => {
        console.log(result);
        return delayedReject("Something went wrong!");
    })
    .catch(error => {
        console.log("Caught error:", error.message);
        return "Recovered from error";
    })
    .then(result => {
        console.log("Final result:", result);
    });

console.log("Promise chain initiated (this appears immediately)");

class EventEmitter {
    constructor() {
        this.events = {};
    }
    
    addEventListener(eventType, handler) {
        if (!this.events[eventType]) {
            this.events[eventType] = [];
        }
        this.events[eventType].push(handler);
    }
    
    removeEventListener(eventType, handler) {
        if (this.events[eventType]) {
            this.events[eventType] = this.events[eventType].filter(h => h !== handler);
        }
    }
    
    dispatchEvent(eventType, eventData = {}) {
        console.log(\`Event triggered: $\{eventType\}\`);
        if (this.events[eventType]) {
            this.events[eventType].forEach(handler => handler(eventData));
        }
    }
}

                return new Rectangle(args[0], args[1]); // width, height
            case 'triangle':

            return Math.PI * this.radius * this.radius;

            return 2 * Math.PI * this.radius;
        }
        
        toString() {
            return \`Circle (r=$\{this.radius\})\`;
        }
    }

        constructor(width, height) {
            this.width = width;
            this.height = height;
        }
        
        getArea() {
            return this.width * this.height;
        }
        
        getPerimeter() {
            return 2 * (this.width + this.height);
        }
        
        toString() {
            return \`Rectangle ($\{this.width\}x$\{this.height\})\`;
        }
    }

        constructor(a, b, c) {
            this.a = a;
            this.b = b;
            this.c = c;

            // Heron's formula
            const s = (this.a + this.b + this.c) / 2;
            return Math.sqrt(s * (s - this.a) * (s - this.b) * (s - this.c));
        }
        
        getPerimeter() {
            return this.a + this.b + this.c;
        }
        
        toString() {
            return \`Triangle ($\{this.a\}, $\{this.b\}, $\{this.c\})\`;
        }
    }
    
    return {
        createShape,
        Circle,
        Rectangle,
        Triangle
    };
